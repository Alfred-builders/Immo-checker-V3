// GET /api/dashboard/activity?range=7j|30j|90j
router.get('/activity', async (req, res) => {
  const workspaceId = req.user!.workspaceId
  const range = Number(req.query.range) || 30  // jours
  const dateFrom = new Date(Date.now() - range * 86400000).toISOString().slice(0,10)

  // 1. Daily entrées/sorties for area chart
  const daily = await query(`
    SELECT date_planifiee::date::text as date,
      count(*) FILTER (WHERE ei.sens = 'entree')::int as entrees,
      count(*) FILTER (WHERE ei.sens = 'sortie')::int as sorties
    FROM mission m
    JOIN edl_inventaire ei ON ei.mission_id = m.id
    WHERE m.workspace_id = $1 AND m.statut != 'annulee'
      AND m.date_planifiee >= $2
    GROUP BY date_planifiee::date ORDER BY 1`, [workspaceId, dateFrom])

  // 2. Monthly trend (last 6 months) for line chart
  const monthly = await query(`
    SELECT to_char(date_trunc('month', date_planifiee), 'Mon') as month,
      count(*)::int as planifiees,
      count(*) FILTER (WHERE statut = 'terminee')::int as terminees
    FROM mission WHERE workspace_id = $1 AND statut != 'annulee'
      AND date_planifiee >= date_trunc('month', now()) - INTERVAL '5 months'
    GROUP BY 1, date_trunc('month', date_planifiee)
    ORDER BY date_trunc('month', date_planifiee)`, [workspaceId])

  // 3. Statut breakdown (all missions) for donut pie
  const statuts = await query(`
    SELECT statut, count(*)::int as count
    FROM mission WHERE workspace_id = $1 AND est_archive = false
    GROUP BY statut`, [workspaceId])

  const statutMap = Object.fromEntries(statuts.rows.map(r => [r.statut, r.count]))
  sendSuccess(res, { daily: daily.rows, monthly: monthly.rows, statuts: statutMap })
})

server/middleware/api-key-auth.ts
export async function verifyApiKey(req, res, next) {
  const auth = req.headers.authorization
  if (!auth?.startsWith('Bearer imk_'))
    return res.status(401).json({ error: 'API key required' })

  const hash = crypto
    .createHash('sha256')
    .update(auth.slice(7))
    .digest('hex')

  const result = await query(
    `UPDATE api_key SET last_used_at = now()
     WHERE key_hash = $1 AND est_active = true
       AND (expires_at IS NULL OR expires_at > now())
     RETURNING workspace_id, scope`,
    [hash]
  )
  if (!result.rows.length)
    return res.status(401).json({ error: 'Invalid or revoked key' })

  req.workspaceId = result.rows[0].workspace_id
  req.apiKeyScope = result.rows[0].scope
  next()
}

  id              UUID PRIMARY KEY DEFAULT uuid_generate_v4(),
  webhook_id      UUID NOT NULL REFERENCES webhook_config(id) ON DELETE CASCADE,
  event_type      VARCHAR(50) NOT NULL,

  attempts        INT         NOT NULL DEFAULT 0,
  last_attempt_at TIMESTAMPTZ,
  response_code   INT,
  created_at      TIMESTAMPTZ NOT NULL DEFAULT now()
);
CREATE INDEX idx_webhook_delivery_status ON webhook_delivery(statut, created_at);
server/services/webhook-service.ts — Dispatch + retry
export async function dispatchWebhook(
  workspaceId: string,
  eventType: string,
  payload: object
) {
  const hooks = await query(
    `SELECT id, url, secret FROM webhook_config
     WHERE workspace_id = $1 AND est_active = true
       AND $2 = ANY(events)`,
    [workspaceId, eventType]
  )
  for (const hook of hooks.rows) {
    const del = await query(
      `INSERT INTO webhook_delivery (webhook_id, event_type, payload)
       VALUES ($1,$2,$3) RETURNING id`,
      [hook.id, eventType, payload]
    )
    // Fire-and-forget — ne bloque PAS la réponse HTTP principale
    setImmediate(() =>
      sendWithRetry(del.rows[0].id, hook.url, hook.secret, eventType, payload)
    )
  }
}

async function sendWithRetry(deliveryId, url, secret, eventType, payload, attempt = 1) {
  const body = JSON.stringify({ event: eventType,
    timestamp: new Date().toISOString(), data: payload })
  const sig = crypto.createHmac('sha256', secret)
    .update(body).digest('hex')

  try {
    const resp = await fetch(url, {
      method: 'POST',
      headers: {
        'Content-Type': 'application/json',
        'X-ImmoChecker-Signature': `sha256=${sig}`,
        'X-ImmoChecker-Event': eventType
      },
      body,
      signal: AbortSignal.timeout(5000)
    })
    await query(
      `UPDATE webhook_delivery SET statut=$1, attempts=$2,
        last_attempt_at=now(), response_code=$3 WHERE id=$4`,
      [resp.ok ? 'success' : 'failed', attempt, resp.status, deliveryId]
    )
    if (!resp.ok && attempt < 3)
      setTimeout(() => sendWithRetry(deliveryId, url, secret, eventType, payload, attempt+1),
        attempt * 30_000)  // 30s → 60s → 90s
  } catch {
    if (attempt < 3)
      setTimeout(() => sendWithRetry(...), attempt * 30_000)
  }
}

~1h · INCLUS dans le plan

Logs livraisons webhook dans l'UI

~2h · INCLUS dans le plan
📈
Invalidation queries dashboard
Vérifier que toutes les mutations (créer mission, annuler, signer EDL) invalident bien dashboard-stats et dashboard-activity dans TanStack Query. Actuellement le refetchInterval est 30s — les stat cards ne se mettent pas à jour après une action.
~30 min · rapide
🚦
Rate limiting par clé plutôt que par IP
Express-rate-limit utilise l'IP par défaut. En production derrière un load balancer, l'IP est celle du proxy. Le keyGenerator doit utiliser Authorization header (le key prefix) pour un rate limit correct par clé API.
~15 min · INCLUS dans le plan
